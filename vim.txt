add to /etc/vim/vimrc /etc/vim/vimrc.tiny and ~/.vimrc:

syntax on
set number
set t_Co=256
set tabstop=4
set softtabstop=4
set shiftwidth=4
set hlsearch
set incsearch
set nowrapscan
set ignorecase smartcase
set shortmess=atI
set autoindent
set cindent
set list
set listchars=tab:>-,trail:-
set cursorline
set nocompatible
set backspace=indent,eol,start
